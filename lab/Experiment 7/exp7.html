Project Structure

Application Code

from flask import Flask app = Flask(__name__) @app.route("/") def home(): return "Hello from CI/CD Pipeline!" app.run(host="0.0.0.0", port=80)

Dockerfile

FROM python:3.10-slim WORKDIR /app COPY . . RUN pip install -r requirements.txt EXPOSE 80 CMD ["python", "app.py"]

Jenkins Pipeline

pipeline { agent any stages { stage('Clone') { steps { git 'repo-url' } } stage('Build') { steps { sh 'docker build -t myapp .' } } stage('Push') { steps { sh 'docker push myapp' } } } }
